    <div class="weather-container">
        <h1>Weather App</h1>
        <div>
            <input type="text" id="locationInput" placeholder="Enter city name..." />
            <button onclick="getWeatherByInput()">Search</button>
            <button onclick="getWeatherByLocation()">Current Location</button>
        </div>
        <div class="weather-display" id="weatherDisplay">
            <small>Weather info will appear here.</small>
        </div>
    </div>

    <script>
        const apiKey = "dbf1bcf3c443631b7cca89ad6bd42fc9";

        async function getWeather(city) {
            const url = `https://api.openweathermap.org/data/2.5/weather?q=${encodeURIComponent(city)}&appid=${apiKey}&units=metric`;
            try {
                const response = await fetch(url);
                if (!response.ok) throw new Error("City not found");
                const data = await response.json();
                displayWeather(data);
                setBackgroundByWeather(data.weather[0].main);
            } catch (error) {
                document.getElementById("weatherDisplay").innerHTML = `<p>${error.message}</p>`;
                document.body.style.backgroundImage = '';
            }
        }

        async function getWeatherByCoords(lat, lon) {
            const url = `https://api.openweathermap.org/data/2.5/weather?lat=${lat}&lon=${lon}&appid=${apiKey}&units=metric`;
            try {
                const response = await fetch(url);
                if (!response.ok) throw new Error("Location not found");
                const data = await response.json();
                displayWeather(data);
                setBackgroundByWeather(data.weather[0].main);
            } catch (error) {
                document.getElementById("weatherDisplay").innerHTML = `<p>${error.message}</p>`;
                document.body.style.backgroundImage = '';
            }
        }

        function getWeatherByInput() {
            const city = document.getElementById("locationInput").value;
            if (city.trim() !== "") {
                getWeather(city);
            } else {
                document.getElementById("weatherDisplay").innerHTML = `<p>Please enter a city name.</p>`;
            }
        }

        function getWeatherByLocation() {
            if (navigator.geolocation) {
                navigator.geolocation.getCurrentPosition(
                    (position) => {
                        getWeatherByCoords(position.coords.latitude, position.coords.longitude);
                    },
                    () => {
                        document.getElementById("weatherDisplay").innerHTML = `<p>Unable to retrieve your location.</p>`;
                    }
                );
            } else {
                document.getElementById("weatherDisplay").innerHTML = `<p>Geolocation is not supported by your browser.</p>`;
            }
        }

        function setBackgroundByWeather(weatherMain) {
            let image = "";
            switch (weatherMain) {
                case "Clear":
                    image = "https://images.unsplash.com/photo-1506744038136-46273834b3fb?auto=format&fit=crop&w=1600&q=80";
                    break;
                case "Clouds":
                    image = "https://images.unsplash.com/photo-1465101162946-4377e57745c3?auto=format&fit=crop&w=1600&q=80";
                    break;
                case "Rain":
                case "Drizzle":
                    image = "https://images.unsplash.com/photo-1444384851176-6e23071c6127?auto=format&fit=crop&w=1600&q=80";
                    break;
                case "Snow":
                    image = "https://images.unsplash.com/photo-1519681393784-d120267933ba?auto=format&fit=crop&w=1600&q=80";
                    break;
                case "Mist":
                case "Fog":
                case "Haze":
                    image = "https://images.unsplash.com/photo-1501594907352-04cda38ebc29?auto=format&fit=crop&w=1600&q=80";
                    break;
                default:
                    image = "https://images.unsplash.com/photo-1465101046530-73398c7f28ca?auto=format&fit=crop&w=1600&q=80";
            }
            document.body.style.backgroundImage = `url('${image}')`;
        }

        function displayWeather(data) {
            document.getElementById("weatherDisplay").innerHTML = `
                <h2>${data.name}, ${data.sys.country}</h2>
                <p>
                    <img class="icon" src="<https://openweathermap.org/img/w/${data.weather>[0].icon}.png" 
                        alt="${data.weather[0].description}" />
                    ${data.weather[0].main} (${data.weather[0].description})
                </p>
                <p>Temperature: ${data.main.temp}°C (min: ${data.main.temp_min}°C, max: ${data.main.temp_max}°C)</p>
                <p>Humidity: ${data.main.humidity}%</p>
                <p>Wind Speed: ${data.wind.speed} m/s</p>
            `;
        }
